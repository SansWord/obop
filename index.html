obop

MongoDB-like Object Operator Translators for JavaScript Array Manipulation

SYNOPSIS

Node.js Environment

var obop = require('obop')();
var src = [ { a: 1 }, { a: 2 }, { a: 3 } ];
var func = obop.where( { a: 2 } );
var out = src.filter(func);
// => [ { a: 2 } ]

Browser Envorinment

<script src="obop.js"></script>
<script>
  var src = [ { a: 1 }, { a: 2 }, { a: 3 } ];
  var func = obop.where( { a: 2 } );
  var out = src.filter(func);
  // => [ { a: 2 } ]
</script>
