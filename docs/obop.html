Class: obop

obop

new obop()

This class provides translation methods for MongoDB-like operator objects.
Properties:
Name Type Description
system.name String Read only. System name: "obop"
system.version String Read only. System version: e.g. "0.0.1"
settings.nop Bool Set this true if you prefer a through function returned, instead of null value returned per default, in a case of no operations applied.
settings.throwError Bool Set this true if you prefer throwing an exception, instead of an Error instance returned per default, in a case of something wrong found.

Examples
// node.js
var obop = require('obop')();
var src = [ { a: 1 }, { a: 2 }, { a: 3 } ];
var func = obop.where( { a: 2 } );

Example
var list = [
   { name: "apple", price: 50 },
   { name: "orange", price: 10 },
   { name: "pineapple", price: 70 },
   { name: "grape", price: 30 }
];

// without obop
var out1 = list.sort(function(a, b) {
    return a.price - b.price;
});
console.log(out1);

// with obop
var order = { price: 1 };
var out2 = list.sort(obop.order(order));
console.log(out2);

    item.sale = true;
    item.price -= 5;
    return item;
});

// with obop

var out2 = list.filter(obop.update(update));
